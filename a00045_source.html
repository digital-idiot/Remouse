116  /**
117  * Constructor.
118  *
119  * <p>
120  * Initializes this <code>EKEProvider</code>. Generates a secret key using
121  * the <i>SHA-256 message digest</i> of the pairing key used for client-server
122  * authentication and the server <code>PublicKey</code>.
123  * </p>
124  * @param pairingKey the pairing key.
125  * @param serverPublicKey public key of the server.
126  */
127  public EKEProvider(byte[] pairingKey, byte[] serverPublicKey) {
128  MessageDigest msgDigest;
129  try {
130  msgDigest = MessageDigest.getInstance(MESSAGE_DIGEST, SECURITY_PROVIDER);
131  mIV = msgDigest.digest(pairingKey);
132  KeyPair keyPair = KeyStoreManager.getKSKeyPair();
133  if (keyPair != null) {
134  X509EncodedKeySpec pubKeySpec = new X509EncodedKeySpec(Base64.decodeBase64(serverPublicKey));
135  KeyFactory keyFactory = KeyFactory.getInstance(KEY_GENERATION_ALGORITHM, SECURITY_PROVIDER);
136  mSecretKey = generateSharedSecret(keyPair.getPrivate(), keyFactory.generatePublic(pubKeySpec));
137  }
138  } catch (NoSuchAlgorithmException | NoSuchProviderException
139  | CertificateException | KeyStoreException | UnrecoverableEntryException
140  | IOException | InvalidKeySpecException e) {
141  e.printStackTrace();
142  }
143  }

171  private SecretKey generateSharedSecret(PrivateKey privateKey, PublicKey publicKey) {
172  try {
173  KeyAgreement keyAgreement = KeyAgreement.getInstance(KEY_AGREEMENT_ALGORITHM, SECURITY_PROVIDER);

227  private Certificate genSelfSignedCert(PublicKey publicKey, PrivateKey privateKey)
228  throws OperatorCreationException, CertificateException {
229  X500Principal issuer = new X500Principal("CN=127.0.0.1");
230  BigInteger serial = BigInteger.valueOf(System.currentTimeMillis());
231  X509v3CertificateBuilder certGen =
232  new JcaX509v3CertificateBuilder(issuer, serial, NOT_BEFORE, NOT_AFTER,
233  issuer, publicKey);
234  X509CertificateHolder certHolder = certGen.build(new JcaContentSignerBuilder(SIGNATURE_ALGORITHM)
235  .setProvider(SECURITY_PROVIDER)
236  .build(privateKey));
237  return new JcaX509CertificateConverter().setProvider(SECURITY_PROVIDER).getCertificate(certHolder);
238  }
