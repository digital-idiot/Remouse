25 /**
26  * Manages the operations on the <code>KeyStore</code>.
27  *
28  * <p>
29  * This class provides a {@link java.security.KeyStore} which stores :
30  * <ul>
31  * <li> The private key ({@link java.security.PrivateKey}).</li>
32  * <li>
33  * A self-signed certificate ({@link java.security.cert.Certificate})
34  * that contains the public key and the digital signature.
35  * </li>
36  * </ul>
37  * The <code>KeyStore</code> is protected by a randomly generated
38  * password.
39  * </p>
40  *
41  * @see project.android.security.EKEProvider
42  * @see java.security.KeyStore
43  * @see java.security.KeyPair
44  * @see java.security.cert.Certificate
45  */

50  private static final String KEY_STORE_TYPE = "pkcs12";
51  private static final String KEY_STORE_ALIAS = "Remouse KeyStore";
52  private static final String KEY_STORE_PASSWORD = generateKSPassword();
53  private static final String KEY_STORE_NAME = sRemouseDir + "/remouse_keystore";

55  /**

77  /**
78  * Method to store the private key and the certificate in the
79  * <code>KeyStore</code>.
80  *
81  * @param privateKey the {@link java.security.PrivateKey} object.
82  * @param cert the {@link java.security.cert.Certificate} object(s).
83  * @throws KeyStoreException
84  * @throws IOException
85  * @throws CertificateException
86  * @throws NoSuchAlgorithmException
87  */
88  void setMasterKey(PrivateKey privateKey, Certificate ...cert) throws KeyStoreException, IOException, CertificateException, NoSuchAlgorithmException {
89  KeyStore.PrivateKeyEntry privateKeyEntry = new KeyStore.PrivateKeyEntry(privateKey, cert);
90  mKeyStore.setEntry(KEY_STORE_ALIAS, privateKeyEntry,new KeyStore.PasswordProtection(KEY_STORE_PASSWORD.toCharArray()));
91  mKeyStore.store(new FileOutputStream(KEY_STORE_NAME), KEY_STORE_PASSWORD.toCharArray());
92  }

94  /**
95  * Returns the private-public key pair.
96  *
97  * @return {@link java.security.KeyPair} object.
98  * @throws IOException
99  * @throws KeyStoreException
100  * @throws CertificateException
101  * @throws NoSuchAlgorithmException
102  * @throws UnrecoverableEntryException
103  */
104  static KeyPair getKSKeyPair() throws IOException, KeyStoreException, CertificateException, NoSuchAlgorithmException, UnrecoverableEntryException {
105  FileInputStream fis = new FileInputStream(KEY_STORE_NAME);
106 
107  KeyStore keyStore = KeyStore.getInstance(KEY_STORE_TYPE);
108  keyStore.load(fis, KEY_STORE_PASSWORD.toCharArray());
109 
110  Key key = keyStore.getKey(KEY_STORE_ALIAS, KEY_STORE_PASSWORD.toCharArray());
111  if(key instanceof PrivateKey) {
112  Certificate certificate = keyStore.getCertificate(KEY_STORE_ALIAS);
113  PublicKey publicKey = certificate.getPublicKey();
114  return new KeyPair(publicKey, (PrivateKey) key);
115  }
116  return null;
117  }
118 
119  private static String generateKSPassword() {
120  String password = sSharedPrefs.getString("Password", null);
121  if(password == null) {
122  SecureRandom random = new SecureRandom();
123  StringBuilder stringBuilder = new StringBuilder(new BigInteger(36, 0, random).toString(Character.MAX_RADIX));
124  while (stringBuilder.length() > 6) {
125  stringBuilder.deleteCharAt(random.nextInt(stringBuilder.length()));
126  }
127  for (int i = 0; i < stringBuilder.length(); i++) {
128  char ch = stringBuilder.charAt(i);
129  if (Character.isLetter(ch) && Character.isLowerCase(ch) && random.nextFloat() < 0.5) {
130  stringBuilder.setCharAt(i, Character.toUpperCase(ch));
131  }
132  }
133  SharedPreferences.Editor editor = sSharedPrefs.edit();
134  password = stringBuilder.toString();
135  editor.putString("Password", password);
136  editor.apply();
137  }
138  return password;
139  }
