108  /**
109  * Overrides the
110  * <code>View.onTouchListener.onTouch(View, MotionEvent)</code>
111  * method of the Android API.
112  *
113  * Called when a touch event is dispatched to a <code>View</code>.
114  * This allows listeners to get a chance to respond before the
115  * target <code>View</code>.
116  *
117  * @param view the view the touch event has been dispatched to.
118  * @param motionEvent the <code>MotionEvent</code> object containing
119  * full information about the event.
120  * @return <code>true</code>, if the listener has consumed the event,<br/>
121  * <code>false</code>, otherwise.
122  */
123  @Override
124  public boolean onTouch(View view, MotionEvent motionEvent) {
125  if (view.getId() == R.id.button_touch) {
126  if (sConnectionAlive.containsValue(true)) processTouch(motionEvent);
127  }
128  if (view.getId() == R.id.button_touch_left) {
129  if (sConnectionAlive.containsValue(true) && motionEvent.getActionMasked() == MotionEvent.ACTION_DOWN) {
130  if (mFirstTouch && (System.currentTimeMillis() - mTouchTime) <= 300) {
131  mFirstTouch = false;
132  sendMouseButtonData("left");
133  } else {
134  mFirstTouch = true;
135  mTouchTime = System.currentTimeMillis();
136  sendMouseButtonData("left");
137  }
138  }
139  }
140  return true;
141  }

186  private void processTouch(MotionEvent event) {
187  switch (event.getActionMasked()) {
188  case MotionEvent.ACTION_DOWN:
189  onSingleClick(event, true);
190  break;
191  case MotionEvent.ACTION_UP:
192  onSingleClick(event, false);
193  break;
194  case MotionEvent.ACTION_MOVE:
195  onMove(event);
196  }
197  }
198 
199  private void sendMouseMovementData(final int x, final int y) {
200  new Thread(new Runnable() {
201  @Override
202  public void run() {
203  if (sConnectionAlive.containsValue(true)) {
204  sSecuredClient.sendData(x, y);
205  }
206  }
207  }).start();
208  }

210  private void onMove(MotionEvent event) {
211  float curMoveX = event.getX();
212  float curMoveY = event.getY();
213  int distanceX = 0, distanceY = 0;
214 
215  if (mLastMoveX != Float.MAX_VALUE && mLastMoveY != Float.MAX_VALUE) {
216  distanceX = (int) (curMoveX - mLastMoveX);
217  distanceY = (int) (curMoveY - mLastMoveY);
218  }
219 
220  sendMouseMovementData(distanceX, distanceY);
221  mLastMoveX = curMoveX;
222  mLastMoveY = curMoveY;
223  }
224 
225  private void onSingleClick(MotionEvent event, boolean down) {
226  if (down) {
227  mLastMoveX = event.getX();
228  mLastMoveY = event.getY();
229  }
230  }
