13 /**
14  * Implementation of the <code>java.lang.Runnable</code> for the thread responsible
15  * for sending keyboard data to the server.
16  *
17  * <p>
18  * This thread maintains a buffer to store the data to be sent to the server. It
19  * is implemented as a <code>java.util.concurrent.LinkedBlockingQueue</code>. The
20  * data contains two fields :
21  * <ul>
22  * <li>
23  * A data field which represents the data to be sent, and
24  * </li>
25  * <li>
26  * A boolean flag which determines if the data represents a special key.
27  * </li>
28  * </ul>
29  * </p>
30  * <p>
31  * The data is inserted to the buffer by the listener methods defined in the
32  * {@link project.android.KeyboardFragment} class. This thread continuously
33  * monitors the buffer to check if data is available. If available, it sends
34  * the data.
35  * </p>
36  * <p>
37  * This thread is started when the keyboard UI is loaded and the
38  * {@link project.android.KeyboardFragment#onCreateView(LayoutInflater, ViewGroup, Bundle)}
39  * method is called. It is stopped by invoking the {@link #setStopFlag()}
40  * method when the keyboard UI is destroyed and the
41  * {@link project.android.KeyboardFragment#onDestroyView()} method is called.
42  * </p>
43  *
44  * @see java.lang.Runnable
45  * @see java.util.concurrent.LinkedBlockingQueue
46  * @see project.android.KeyboardFragment
47  * @see project.android.KeyboardFragment#onCreateView(LayoutInflater, ViewGroup, Bundle)
48  * @see project.android.KeyboardFragment#onDestroyView()
49  * @see #setStopFlag()
50  */
51 public class KeyboardThread implements Runnable {

92  /**
93  * Performs the sending of keyboard data in the buffer to the server.
94  *
95  * <p>
96  * It checks if data is available in the buffer.
97  * <ul>
98  * <li>If available, it sends the data.</li>
99  * <li>Otherwise, it waits for the data to be available.</li>
100  * </ul>
101  * </p>
102  */
103  @Override
104  public void run() {
105  while(!mStopFlag) {
106  try {
107  Pair<String, Boolean> data = mBuffer.poll(POLL_TIMEOUT, TIME_UNIT);
108  if(data != null) {
109  if (data.second) {
110  sSecuredClient.sendData(data.first);
111  } else sSecuredClient.sendData("Key", data.first);
112 // Log.d("Keyboard sent", data.first + "__>" + data.first.length());
113  }
114  } catch (InterruptedException e) { e.printStackTrace(); }
115  }
116  }
