64  * <ul>
65  * <li>

74  public void run() {

76  while (!mStopFlag) {
77  recieveData();
78  }

94  private void recieveData() throws IOException {
95  BufferedReader in = sSecuredClient.getSocketReader();
96  if(!in.ready()) return;
97  String s = mEKEProvider.decryptString(in.readLine());
98  if(s != null) {
99  if (s.equals("Stop")) {
100  sConnectionAlive.put(mAddress, false);
101  mStopFlag = true;
102  return;
103  }
104  sConnectionAlive.put(mAddress, s.equals("1"));
105  }
106  if (s == null || !sConnectionAlive.get(mAddress)) {
107  displayError(mActivity);
108  } else {
109  Intent intent = new Intent(mActivity, NetworkService.class);
110  intent.putExtra("Server", mAddress);
111  mActivity.startService(intent);
112  }
113  }
