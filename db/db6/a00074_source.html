12 /**
13  * Class representing the <code>Activity</code> for app settings.
14  *
15  * <p>
16  * It displays the app settings and enables customization
17  * for the users. There are two customizations that are
18  * present :
19  * <ol>
20  * <li>3D Mouse Sensitivity.</li>
21  * <li>2D Mouse Sensitivity.</li>
22  * </ol>
23  * Each of these can have three possible values - : <i>"Low"</i>,
24  * <i>"Medium"</i> and <i>"High"</i>. The default value is
25  * <i>"Medium"</i>.
26  * </p>
27  * <p>
28  * This class is made separate from the {@link MainActivity}
29  * to enable the <i>"Settings"</i> to be opened from any
30  * fragment of the <code>MainActivity</code>. This avoids
31  * overlapping of other fragments - that are already rendered -
32  * by the {@link SettingsFragment}.
33  * </p>
34  * <p>
35  * It also implements the
36  * <code>SharedPreferences.OnSharedPreferenceListener</code>
37  * of the Android API. The
38  * {@link #onSharedPreferenceChanged(SharedPreferences, String)}
39  * method listens to any changes made in the app settings.
40  * </p>
41  *
42  * @see project.android.MainActivity
43  * @see project.android.SettingsFragment
44  * @see #onSharedPreferenceChanged(SharedPreferences, String)
45  */
46 public class SettingsActivity extends AppCompatActivity implements SharedPreferences.OnSharedPreferenceChangeListener {

75  /**
76  * Overrides the
77  * <code>
78  * OnSharedPreferenceChangeListener.onSharedPreferenceChanged
79  * (SharedPreferences, String)
80  * </code>
81  * method of the Android API.
82  *
83  * @param sharedPreferences the <code>SharedPreferences</code>
84  * that received the change.
85  * @param key the key of the preference that was changed, added,
86  * or removed.

88  @Override
89  public void onSharedPreferenceChanged(SharedPreferences sharedPreferences, String key) {
90  if(key.equals("sensitivity_3D")) {
91  sMouseSensitivity3d = Float.parseFloat(sharedPreferences.getString(key, "2000.0"));
92  } else {
93  sMouseSensitivity2d = Float.parseFloat(sharedPreferences.getString(key, "1.5"));
94  }
95  SharedPreferences.Editor editor = sSharedPrefs.edit();
96  editor.putFloat("2d", sMouseSensitivity2d);
97  editor.putFloat("3d", sMouseSensitivity3d);
98  editor.apply();
99  }
