58  *
59  * <p>
60  * Called when there is a new sensor event. It is also called if there
61  * is a reading available from a sensor with the exact same sensor

63  * </p>
64  *
65  * @param event the <code>SensorEvent</code> object containing full
66  * information about the event.
67  */
68  @Override
69  public void onSensorChanged(SensorEvent event) {
70 
71  // it is a good practice to check that we received the proper event
72  if (event.sensor.getType() == Sensor.TYPE_GYROSCOPE) {
73 
74  // This timestamps delta rotation to be multiplied by the current rotation
75  if (mTimestamp != 0) {
76  final float dT = (event.timestamp - mTimestamp) * NS2S;
77  // Axis of the rotation sample, not normalized yet.
78  float axisX = event.values[0];
79  float axisY = event.values[1];
80  float axisZ = event.values[2];
81 
82  // Calculate the angular speed of the sample
83 
84  double gyroscopeRotationVelocity = Math.sqrt(axisX * axisX + axisY * axisY + axisZ * axisZ);
85 
86  if (gyroscopeRotationVelocity > EPSILON) {
87  axisX /= gyroscopeRotationVelocity;
88  axisY /= gyroscopeRotationVelocity;
89  axisZ /= gyroscopeRotationVelocity;
90  }
91 
92  double thetaOverTwo = gyroscopeRotationVelocity * dT / 2.0f;
93  double sinThetaOverTwo = Math.sin(thetaOverTwo);
94  double cosThetaOverTwo = Math.cos(thetaOverTwo);
95  deltaQuaternion.setX((float) (sinThetaOverTwo * axisX));
96  deltaQuaternion.setY((float) (sinThetaOverTwo * axisY));
97  deltaQuaternion.setZ((float) (sinThetaOverTwo * axisZ));
98  deltaQuaternion.setW(-(float) cosThetaOverTwo);
99 
100  synchronized (synchronizationToken) {
101  deltaQuaternion.multiplyByQuat(currentOrientationQuaternion, currentOrientationQuaternion);
102  }
103 
104  mCorrectedQuaternion.set(currentOrientationQuaternion);
105 
106  mCorrectedQuaternion.setW(-mCorrectedQuaternion.getW());
107 
108  synchronized (synchronizationToken) {
109  // Set the rotation matrix as well to have both representations
110  SensorManager.getRotationMatrixFromVector(currentOrientationRotationMatrix.matrix,
111  mCorrectedQuaternion.array());
112  }
113  if(sConnectionAlive.containsValue(true) && sMouseAlive) {
114  new Thread(new Runnable() {
115  @Override
116  public void run() {
117  sSecuredClient.sendData(mCorrectedQuaternion, mIsInitQuat);
118  mIsInitQuat = false;
119  }
120  }).start();
121  } else sensorStop();
122  }
123  mTimestamp = event.timestamp;
124  }
125  }
