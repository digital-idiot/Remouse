64  /**
65  * Performs the broadcast receiving operation.
66  *
67  * <ul>
68  * <li>
69  * Checks whether a stop signal has been broadcast by the server. If so,
70  * removes the {@link ServerInfo} object from the <code>ListView</code> in
71  * the {@link project.android.ConnectionFragment}.
72  * </li>
73  * <li>
74  * Constructs a {@link ServerInfo} object from the data obtained from the
75  * <code>DatagramPacket</code> using <code>com.google.Gson</code> API and
76  * updates the <code>ListView</code> in the
77  * {@link project.android.ConnectionFragment} accordingly.
78  * </li>
79  * </ul>
80  *
81  * @see project.android.ConnectionFragment
82  */
83  @Override
84  public void run() {
85  try {
86  mLock.acquire();
87  if (mDatagramSocket == null) {
88  mDatagramSocket = new DatagramSocket(UDP_PORT);
89  mDatagramSocket.setBroadcast(true);
90  }
91  } catch (IOException e) { e.printStackTrace(); }
92 
93  while (mDatagramSocket != null) {
94  DatagramPacket datagramPacket = null;
95  try {
96  if (!mLock.isHeld()) mLock.acquire();
97  datagramPacket = new DatagramPacket(new byte[ServerInfo.SERVER_INFO_LENGTH],
98  ServerInfo.SERVER_INFO_LENGTH);
99  mDatagramSocket.receive(datagramPacket);
100  mLock.release();
101  } catch (IOException ignored) {}
102  String gsonString = new String(datagramPacket.getData());
103  gsonString = gsonString.substring(0, gsonString.indexOf("}") + 1);
104  ServerInfo serverInfo = new Gson().fromJson(gsonString, ServerInfo.class);
105  String serverAddress = datagramPacket.getAddress().toString().substring(1);
106  serverInfo.setServerAddress(serverAddress);
107 
108  checkAvailableDevices(serverInfo, serverAddress);
109  }
110  }
111  private void checkAvailableDevices(final ServerInfo serverInfo, final String serverAddress) {
112 
113  if (serverInfo.getStopFlag()) {
114  mServerAddress.remove(serverAddress);
115  mActivity.runOnUiThread(new Runnable() {
116  @Override
117  public void run() {
118  Toast.makeText(mActivity, "Server " + serverInfo.getAddress() + " stopped!",
119  Toast.LENGTH_SHORT).show();
120  mConnectionFragment.removeItem(serverInfo);
121  mConnectionFragment.dismissAlertDialog();
122  }
123  });
124  } else {
125  if (mServerAddress.isEmpty() || !mServerAddress.contains(serverAddress)) {
126  mServerAddress.add(serverAddress);
127  mActivity.runOnUiThread(new Runnable() {
128  @Override
129  public void run() {
130  mConnectionFragment.addItem(serverInfo);
131  sConnectionAlive.put(serverAddress, false);
132  }
133  });
134  }
135  }
